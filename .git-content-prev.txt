'not-allowed' : 'pointer' }}>
                      <Download size={14} />

                </div>
              </div>

          })}
        </div>
      )}

      {/* Video Lightbox */}
      {activeVideoUrl && createPortal(
        <div role="dialog" aria-modal="true" aria-label="в”ҳГ в•Әв”Өв•Әв•‘в”ҳВ„ в•ӘШҜв”ҳВ„в”ҳВҒв”ҳГЁв•ӘВ»в”ҳГЁв”ҳГӘ" style={{

          backdropFilter: 'blur(8px)', zIndex: 9999,

            role="dialog"

            aria-labelledby="edit-content-title"

              position: 'fixed', inset: 0, zIndex: 9999,

                  <h2 id="edit-content-title" style={{ margin: 0, fontSize: '1.25rem', fontWeight: 700, color: 'var(--text-primary)' }}>в•ӘШІв•Әв•Јв•ӘВ»в”ҳГЁв”ҳВ„ в•ӘШҜв”ҳВ„в”ҳГ в•ӘШөв•ӘШІв”ҳГӘв”ҳГ«</h2>

              <form onSubmit={handleEditContentSubmit} style={{ display: 'grid', gap: '18px' }}>

                    </select>
                  </div>
                  <div style={{ display: 'flex', flexDirection: 'column', justifyContent: 'space-between' }}>
                    <label htmlFor="edit-download-protection" style={{ display: 'block', fontSize: '0.85rem', fontWeight: 600, color: 'var(--text-secondary)', marginBottom: '8px' }}>в•ӘШөв”ҳГ в•ӘШҜв”ҳГЁв•ӘШұ в•ӘШҜв”ҳВ„в•ӘШІв”ҳВҶв•Әв–“в”ҳГЁв”ҳВ„</label>

                      <input
                        id="edit-download-protection"
                        type="checkbox"
                        checked={editContentForm.downloadProtected}
                        onChange={(e) => setEditContentForm(prev => ({ ...prev, downloadProtected: e.target.checked }))}
                      />
                      <span style={{ color: 'var(--text-primary)', fontSize: '0.9rem' }}>в”ҳГ в”ҳВҶв•Әв•Ј в•ӘШҜв”ҳВ„в•ӘШІв”ҳВҶв•Әв–“в”ҳГЁв”ҳВ„</span>

                  <label htmlFor="edit-file" style={{ display: 'block', fontSize: '0.85rem', fontWeight: 600, color: 'var(--text-secondary)', marginBottom: '8px' }}>в•ӘШҜв•Әв”Ӯв•ӘШІв•ӘШ°в•ӘВ»в•ӘШҜв”ҳВ„ в•ӘШҜв”ҳВ„в”ҳГ в”ҳВ„в”ҳВҒ</label>
                  <input
                    id="edit-file"
                    type="file"
                    onChange={(e) => setEditContentForm(prev => ({ ...prev, file: e.target.files?.[0] || null }))}
                    style={{ width: '100%', padding: '10px 14px', borderRadius: 'var(--radius-sm)', border: '1px solid var(--input-border)', background: 'var(--input-bg)', color: 'var(--text-primary)' }}
                  />

                {editUploading && (
                  <div style={{ marginTop: '8px' }}>
                    <div style={{ display: 'flex', justifyContent: 'space-between', fontSize: 'var(--text-xs)', color: 'var(--text-secondary)', marginBottom: '4px' }}>
                      <span>в•ӘШҙв•ӘШҜв•Әв–’в”ҳГЁ в•Әв–’в”ҳВҒв•Әв•Ј в•ӘШҜв”ҳВ„в”ҳГ в”ҳВ„в”ҳВҒ...</span>

                    </div>
                  </div>
                )}

                <div style={{ display: 'flex', justifyContent: 'space-between', gap: '12px', flexWrap: 'wrap' }}>
                  <button
                    type="button"
                    onClick={() => setEditContentItem(null)}
                    style={{ background: 'transparent', border: '1px solid var(--border-color)', padding: '10px 18px', borderRadius: 'var(--radius-sm)', color: 'var(--text-secondary)', cursor: 'pointer' }}
                  >
                    в•ӘШӯв”ҳВ„в•Әв•‘в•ӘШҜв•ӘШ©
                  </button>
                  <button
                    type="submit"
                    disabled={editUploading}
                    style={{ background: 'var(--gradient-primary)', border: 'none', color: 'white', padding: '10px 18px', borderRadius: 'var(--radius-sm)', cursor: 'pointer' }}
                  >
                    {editUploading ? 'в•ӘШҙв•ӘШҜв•Әв–’в”ҳГЁ в•ӘШҜв”ҳВ„в•ӘШөв”ҳВҒв•Әв••...' : 'в•ӘШөв”ҳВҒв•Әв•• в•ӘШҜв”ҳВ„в•ӘШІв•Әв•Јв•ӘВ»в”ҳГЁв”ҳВ„в•ӘШҜв•ӘШІ'}
                  </button>
                </div>
              </form>
            </div>
          </div>
        )}
    </div>
  );
